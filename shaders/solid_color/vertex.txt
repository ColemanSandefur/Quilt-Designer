#version 140

in vec2 position;
in vec4 color;
in mat4 model;
in uint tex_id;

out vec4 v_color;
flat out uint v_tex_id;
out vec2 v_tex_coords;

uniform mat4 view;
uniform mat4 projection;

void main() {
    v_color = color;
    v_tex_id = tex_id;
    v_tex_coords = position;

    gl_Position = projection * view * model * vec4(position, 1.0, 1.0);
}
